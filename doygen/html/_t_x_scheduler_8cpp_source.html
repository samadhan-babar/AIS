75void TXScheduler::reportTXStatus()
76{
77 bool hwSwitchOff = bsp_is_tx_disabled();
78 bool softSwitch = Configuration::instance().isTXEnabled();
79// bool hasStation = Configuration::instance().isStationDataProvisioned();
80 bool hasStation = true; //Added by ishan
81 bool status = hwSwitchOff ? false : (softSwitch && hasStation);
82
83 Event *e = EventPool::instance().newEvent(PROPR_NMEA_SENTENCE);
84 if ( !e )
85 return;
86
87 sprintf(e->nmeaBuffer.sentence, "$PAITXCFG,%d,%d,%d,%d,%d*", bsp_is_tx_hardwired(), !hwSwitchOff, softSwitch, hasStation, status);
88 Utils::completeNMEA(e->nmeaBuffer.sentence);
89 EventQueue::instance().push(e);
90}

92bool TXScheduler::isTXAllowed()
93{
94 bool hwSwitchOff = bsp_is_tx_disabled();
95 if ( hwSwitchOff )
96 return false;
97
98 bool softSwitch = Configuration::instance().isTXEnabled();
99 bool hasStation = Configuration::instance().isStationDataProvisioned();
100
101 return softSwitch && hasStation;
102}

104void TXScheduler::processEvent(const Event &e)
105{
106 switch(e.type)
107 {
108 case GPS_FIX_EVENT:
109 {
110 mLastGPSFix = e.gpsFix;
111
112
113 // We do not schedule transmissions if the ChannelManager is not sure what channels are in use yet
114 if ( !ChannelManager::instance().channelsDetermined() )
115 return;
116
117 // A class B transponder only transmits when its internal GPS is working, so we tie these to GPS updates.
118 if ( !RadioManager::instance().initialized() || mUTC == 0 )
119 return;
120
121 if ( !isTXAllowed() )
122 return;
123
124 // Using a moving average of SOG to determine transmission rate
125 static float alpha = 0.2;
126 mAvgSpeed = mAvgSpeed * (1.0 - alpha) + mLastGPSFix.speed * alpha;
127
128 if ( mUTC - mLast18Time > positionReportTimeInterval() )
129 {
130 queueMessage18(mPositionReportChannel);
131 // Our next position report should be on the other channel
132 mPositionReportChannel = RadioManager::instance().alternateChannel(mPositionReportChannel);
133 mLast18Time = mUTC;
134 }
135
136 if ( mUTC - mLast24Time > MSG_24_TX_INTERVAL )
137 {
138 queueMessage24(mStaticDataChannel);
139 // Our next static data report should be on the other channel
140 mStaticDataChannel = RadioManager::instance().alternateChannel(mStaticDataChannel);
141 mLast24Time = mUTC;
142 }
143
144 break;
145 }
146 case CLOCK_EVENT:
147 {
148 // This is reliable and independent of GPS update frequency which could change to something other than 1Hz
149 if ( mUTC == 0 )
150 {
151 // Don't start transmitting right away
152 mLast18Time = e.clock.utc - MAX_MSG_18_TX_INTERVAL/2;
153 mLast24Time = e.clock.utc - MSG_24_TX_INTERVAL/2;
154 }
155
156 mUTC = e.clock.utc;
157
158 break;
159 }
160
161 break;
162 case INTERROGATION_EVENT:
163 if ( e.interrogation.messageType == 18 )
164 queueMessage18(e.interrogation.channel);
165
166 if ( e.interrogation.messageType == 24 )
167 queueMessage24(e.interrogation.channel);
168 break;
169 default:
170 break;
171 }
172
173}

186void TXScheduler::queueMessage18(VHFChannel channel)
187{
188#if REPORT_TX_SCHEDULING
189 char sentence[48];
190#endif
191
192 // If we don't have valid station data we don't do anything
193 if ( mStationData.magic != STATION_DATA_MAGIC )
194 return;
195
196 TXPacket *p1 = TXPacketPool::instance().newTXPacket(channel);
197 if ( !p1 )
198 {
199#if REPORT_TX_SCHEDULING
200 sprintf(sentence, "$PAISCHTX,18,%d*", TX_ALLOC_ERROR);
201 sendNMEASentence(sentence);
202#endif
203 return;
204 }
205
206 AISMessage18 msg;
207 msg.latitude = mLastGPSFix.lat;
208 msg.longitude = mLastGPSFix.lng;
209 msg.sog = mLastGPSFix.speed;
210 msg.cog = mLastGPSFix.cog;
211 msg.utc = mLastGPSFix.utc;
212 msg.encode (mStationData, *p1);
213
214 RadioManager::instance ().scheduleTransmission (p1);
215}

258time_t TXScheduler::positionReportTimeInterval()
259{
260 // As a class B "CS" transponder, we transmit at a rate based on our speed (2 knots is the threshold)
261 if ( mAvgSpeed < 2.0 )
262 return MAX_MSG_18_TX_INTERVAL;
263
264 return MIN_MSG_18_TX_INTERVAL;
