79 bool push(T &Element)
80 {
81 int nextElement = (mWritePosition + 1) % mSize;
82 if ( isSafeToWrite(nextElement) )
83 {
84 mBuffer[mWritePosition] = Element;
85 mWritePosition = nextElement;
86 return true;

100 if ( empty() )
101 return false;
102
103 Element = mBuffer[mReadPosition];
104 mReadPosition = (mReadPosition + 1) % mSize;
105 return true;
106 }
