75void AISMessage::addBits(uint8_t *bitVector, uint16_t &size, uint32_t value, uint8_t numBits)
76{
77 ASSERT(numBits > 0 && numBits <= 32);
78 uint16_t pos = size;
79 for ( uint8_t bit = 0; bit < numBits; ++bit, value >>= 1 ) {
80 bitVector[pos + numBits-bit-1] = value & 1;
81 }
82
83 size += numBits;
84}

86void AISMessage::putBits(uint8_t *bitVector, uint32_t value, uint8_t numBits)
87{
88 // This is used for HDLC framing
89 uint16_t pos = 0;
90 for ( uint8_t bit = 0; bit < numBits; ++bit, value >>= 1 ) {
91 bitVector[pos++] = value & 0x01;
92 }
93}

95void AISMessage::reverseEachByte(uint8_t *bitVector, uint16_t size)
96{
97 ASSERT(size % 8 == 0);
98 for ( uint16_t i = 0; i < size; i += 8 ) {
99 for ( uint8_t j = 0; j < 4; ++j ) {
100 swap(bitVector[i+j], bitVector[i+7-j]);
101 }
102 }
103}

105void AISMessage::addString(uint8_t *bitVector, uint16_t &size, const string &value, uint8_t maxChars)
106{
107 ASSERT(value.length() <= maxChars);
108 ASSERT(maxChars < 30); // There should be no application for such long strings here
109 char s[30];
110 memset(s, 0, sizeof s);
111 strncpy(s, value.c_str(), value.length());
112
113 uint8_t buffer[32];
114 for ( uint8_t c = 0; c < maxChars; ++c ) {
115 uint8_t byte = s[c] >= 64 ? s[c]-64 : s[c];
116 buffer[c] = byte;
117 }
118
119 for ( uint8_t c = 0; c < maxChars; ++c )
120 addBits(bitVector, size, buffer[c], 6);
121}

123void AISMessage::payloadToBytes(uint8_t *bitVector, uint16_t numBits, uint8_t *byteArray)
124{
125 for ( uint16_t i = 0; i < numBits; i += 8 ) {
126 uint8_t byte = 0;
127 for ( uint8_t b = 0; b < 8; ++b ) {
128 byte |= (bitVector[i+b] << b);
129 }
130 byteArray[i/8] = byte;
131 }
132}

134void AISMessage::finalize(uint8_t *payload, uint16_t &size, TXPacket &packet)
135{
136 // Nothing we send exceeds 256 bits, including preambles and such. 40 bytes is more than enough.
137 uint8_t bytes[40];
138
139 // CRC-CCITT calculation
140 payloadToBytes(payload, size, bytes);
141 uint16_t crc = Utils::crc16(bytes, size/8);
142 uint8_t crcL = crc & 0x00ff;
143 uint8_t crcH = (crc & 0xff00) >> 8;
144 addBits(payload, size, crcL, 8);
145 addBits(payload, size, crcH, 8);
146 payloadToBytes(payload, size, bytes);
147
148 // Encoding for transmission
149 reverseEachByte(payload, size);
150 bitStuff(payload, size);
151 constructHDLCFrame(payload, size);
152 nrziEncode(payload, size, packet);
153 packet.pad();
154}

156void AISMessage::bitStuff(uint8_t *buff, uint16_t &size)
157{
158 uint16_t numOnes = 0;
159 for ( uint16_t i = 0; i < size; ++i )
160 {
161 switch(buff[i])
162 {
163 case 0:
164 numOnes = 0;
165 break;
166 case 1:
167 ++numOnes;
168 if ( numOnes == 5 )
169 {
170 // Insert a 0 right after this one
171 memmove(buff + i + 2, buff + i + 1, size-i-1);
172 buff[i+1] = 0;
173 ++size;
174 }
175 break;
176 default:
177 ASSERT(false);
178 }
179 }
180}

182void AISMessage::constructHDLCFrame(uint8_t *buff, uint16_t &size)
183{
184 /*
185 * As a class B "CS" transponder, we don't transmit a full ramp byte because
186 * we have to listen for a few bits into each slot for Clear Channel Assessment.
187 * Also, this implementation only adds 3 bits for ramp down, just in case
188 * our TX bit clock is a little lazy. Not what the ITU standard says, but no
189 * reasonable receiver should care about ramp-down bits. It's only what goes
190 * between the 0x7E markers that counts.
191 */
192
193 // Make room for 35 bits at the front
194 memmove(buff+35, buff, size);
195 size += 35;
196 putBits(buff, 0xFF, 3); // 3 ramp bits. That's all we can afford.
197 putBits(buff+3, 0b010101010101010101010101, 24); // 24 training bits (ramp will actually continue during the first 1-2)
198 putBits(buff+27, 0x7e, 8); // HDLC start flag
199
200 // Now append the end marker and ramp-down bits
201 addBits(buff, size, 0x7e, 8); // HDLC stop flag
202 addBits(buff, size, 0x00, 3); // Ramp down
203}

205void AISMessage::nrziEncode(uint8_t *buff, uint16_t &size, TXPacket &packet)
206{
207 uint8_t prevBit = 1; // Arbitrarily starting with 1
208 packet.addBit(prevBit);
209
210 for ( uint16_t i = 0; i < size; ++i )
211 {
212 if ( buff[i] == 0 )
213 {
214 packet.addBit(!prevBit);
215 prevBit = !prevBit;
216 }
217 else
218 {
219 packet.addBit(prevBit);
220 }
221 }
222
223 // The TXPacket is now populated with the sequence of bits that need to be sent
224}

259bool AISMessage15::decode(const RXPacket &packet)
260{
261 mType = packet.messageType();
262 mRI = packet.repeatIndicator();
263 mMMSI = packet.mmsi();
264
265 uint8_t bit = 40;
266
267 // A message 15 has up to 3 targets
268 for ( uint8_t i = 0; i < 3; ++i ) {
269 if ( bit >= packet.size()-16 )
270 break;
271 targets[i].mmsi = packet.bits(bit, 30);
272 bit += 30;
273 targets[i].messageType = (uint8_t)packet.bits(bit, 6);
274 bit += 6;
275
276 // Ignore slot offset plus 2 spare bits per record
277 bit += 14;
278 }
279
280 return true; // Would we ever return false?
281}

435void AISMessage24B::encode(const StationData &station, TXPacket &packet)
436{
437 AISMessage::encode(station, packet);
438
439 packet.setMessageType("24B");
440 uint8_t payload[MAX_AIS_TX_PACKET_SIZE];
441 uint16_t size = 0;
442 uint32_t value;
443
444 value = mType;
445 addBits(payload, size, value, 6); // Message type
446
447 value = mRI;
448 addBits(payload, size, value, 2); // Repeat Indicator
449
450 value = mMMSI;
451 addBits(payload, size, value, 30); // MMSI
452
453 value = 1;
454 addBits(payload, size, value, 2); // Part number (1 for 24B)
455
456 value = station.type;
457 addBits(payload, size, value, 8); // Type of ship
458
459 addString(payload, size, "", 7); // Vendor information
460
461 addString(payload, size, station.callsign, 7); // Call sign
462
463
464 if ( station.len == 0 || station.beam == 0 )
465 {
466 value = 0;
467 }
468 else
469 {
470 uint16_t A,B,C,D;
471 C = station.portOffset;
472 D = station.beam - C;
473 A = station.bowOffset;
474 B = station.len - A;
475
476 if ( D > 63 )
477 D = 63;
478
479 if ( C > 63 )
480 C = 63;
481
482 if ( B > 511 )
483 B = 511;
484
485 if ( A > 511 )
486 A = 511;
487
488 value = D | (C << 6) | (B << 12) | (A << 21);
489 }
490
491 addBits(payload, size, value, 30); // Dimension information
492
493 value = 1;
494 addBits(payload, size, value, 4); // We are using GPS only
495
496 value = 0;
497 addBits(payload, size, value, 2); // Spare bits
498
499 finalize(payload, size, packet);
500}
