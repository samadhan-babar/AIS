44void Transceiver::configure()
45{
46 Receiver::configure();
47
48 // Anything transmitter specific goes here
49 SET_PROPERTY_PARAMS p;
50
51 p.Group = 0x20;
52 p.NumProperties = 1;
53 p.StartProperty = 0x00;
54 p.Data[0] = 0x20 | 0x08 | 0x03; // Synchronous direct mode from GPIO 1 with 2GFSK modulation
55 sendCmd(SET_PROPERTY, &p, 4, NULL, 0);
56
61 p.Group = 0x22;
62 p.NumProperties = 1;
63 p.StartProperty = 0x06;
64 p.Data[0] = 0xff;
65 sendCmd(SET_PROPERTY, &p, 6, NULL, 0);
66
67 /*
68 * AIS uses a slightly different Gaussian shape than default GMSK in the Si4463.
69 * This sets the TX filter coefficients for a BT of 0.4 as per ITU specification.
70 *
71 * https://www.silabs.com/community/wireless/proprietary/forum.topic.html/si4463_gmsk_spectrum-ojkN
72 */
73 uint8_t data[] = { 0x52, 0x4f, 0x45, 0x37, 0x28, 0x1a, 0x10, 0x09, 0x04 };
74 p.Group = 0x22;
75 p.NumProperties = 9;
76 p.StartProperty = 0x0f;
77 memcpy(p.Data, data, sizeof data);
78 sendCmd(SET_PROPERTY, &p, 12, NULL, 0);
79
80 // These resulted from characterization with default BOM
81 pa_params pwr;
82 switch(mPartNumber)
83 {
84 case 0x4463:
85 pwr.pa_mode = 0x48;
86 pwr.pa_level = 0x10;
87 pwr.pa_bias_clkduty = 0x00;
88 break;
89 case 0x4467:
90 pwr.pa_mode = 0x48;
91 pwr.pa_level = 0x1C;
92 pwr.pa_bias_clkduty = 0x00;
93 break;
94 default:
95 pwr.pa_mode = 0x48;
96 pwr.pa_level = 0x1C;
97 pwr.pa_bias_clkduty = 0x00;
98 break;
99 }
100
101 setTXPower(pwr);
102}

116void Transceiver::transmitCW(VHFChannel channel)
117{
118 startReceiving(channel, false);

187void Transceiver::onBitClock()
188{
189 if ( gRadioState == RADIO_RECEIVING )
190 {
191 Receiver::onBitClock();
192 /*
193 We start transmitting a packet if:
194 - We have a TX packet assigned
195 - Transmission is enabled
196 - We are at bit CCA_SLOT_BIT+1 (after obtaining an RSSI level)
197 - The TX packet's transmission channel is our current listening channel
198 - The RSSI is within 6dB of the noise floor for this channel
199 - It's been at least MIN_TX_INTERVAL seconds since our last transmission
200 */
201
202 if ( !mTXPacket )
203 return;
204
205
206 if ( mTXPacket->isTestPacket() )
207 {
208 // Test packets are sent immediately. Presumably, we're firing into a dummy load ;)
209 startTransmitting();
210 }
211 else if ( !TXScheduler::instance().isTXAllowed() )
212 {
213 // Transmission has been disabled since scheduling, so don't transmit!
214 TXPacketPool::instance().deleteTXPacket(mTXPacket);
215 mTXPacket = NULL;
216 }
217 else if ( mUTC && mUTC - mTXPacket->timestamp() >= MIN_MSG_18_TX_INTERVAL )
218 {
219 // The packet is way too old. Discard it.
220 DBG("Packet is too old. Discarding!\r\n");
221#if REPORT_TX_SCHEDULING
222 Event *e = EventPool::instance().newEvent(PROPR_NMEA_SENTENCE);
223 if ( e )
224 {
225 sprintf(e->nmeaBuffer.sentence, "$PAISCHTX,%s,%d*", mTXPacket->messageType(), TX_PACKET_TOO_OLD);
226 Utils::completeNMEA(e->nmeaBuffer.sentence);
227 EventQueue::instance().push(e);
228 }
229#endif
230 TXPacketPool::instance().deleteTXPacket(mTXPacket);
231 mTXPacket = NULL;
232 }
233 else if ( mUTC - mLastTXTime < MIN_TX_INTERVAL )
234 {
235 DBG("It's not time to transmit yet\r\n");
236 // It's not time to transmit yet
237 return;
238 }
239 else if ( mUTC && mSlotBitNumber == CCA_SLOT_BIT && mTXPacket->channel() == mChannel )
240 {
241 // It has already been sampled during Receiver::onBitClock();
242 int rssi = mRXPacket->rssi();
243 int nf = NoiseFloorDetector::instance().getNoiseFloor(AIS_CHANNELS[mChannel].designation);
244 if ( rssi <= nf + TX_CCA_HEADROOM )
245 {
246 startTransmitting();
247 }
248
249 }
250 }
251 else if ( mTXPacket )
252 {
253 if ( mTXPacket->eof() )
254 {
255 mLastTXTime = mUTC;
256 startReceiving(mChannel, true);
257 gRadioState = RADIO_RECEIVING;
258 reportTXEvent();
259 TXPacketPool::instance().deleteTXPacket(mTXPacket);
260 mTXPacket = NULL;
261 }
262 else
263 {
264 uint8_t bit = mTXPacket->nextBit();
265 if ( bit )
266 HAL_GPIO_WritePin(mDataPort, mDataPin, GPIO_PIN_SET);
267 else
268 HAL_GPIO_WritePin(mDataPort, mDataPin, GPIO_PIN_RESET);
269
279 if ( mTXPacket->canRampDown() )
280 HAL_GPIO_WritePin(PA_BIAS_PORT, PA_BIAS_PIN, GPIO_PIN_RESET);
281 }
282 }
283 else
284 {
285 // We're transmitting carrier wave, do nothing. This is only done during XO trimming.
286 }
287}

349void Transceiver::reportTXEvent()
350{
351 Event *e = EventPool::instance().newEvent(PROPR_NMEA_SENTENCE);
352 if ( !e )
353 return;
354
355 snprintf(e->nmeaBuffer.sentence, sizeof e->nmeaBuffer.sentence, "$PAITX,%c,%s*", AIS_CHANNELS[mTXPacket->channel()].designation, mTXPacket->messageType());
356 Utils::completeNMEA(e->nmeaBuffer.sentence);
357 EventQueue::instance().push(e);
358 bsp_tx_led_off();
