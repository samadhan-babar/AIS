48void NoiseFloorDetector::report(char channel, uint8_t rssi)
49{
50#if BOARD_REV == 61 or BOARD_REV == 52
51 if ( rssi < 0x12 ) // Not realistic, likely a bug
52#elif BOARD_REV == 109
53 if ( rssi < 0x24 )
54#else
55 if ( rssi < 0x10 ) // Not realistic, likely a bug
56#endif
57 return;
58
59 if ( channel == 'A' )
60 {
61 mAFloor = min(mAFloor, rssi);
62 }
63 else
64 {
65 mBFloor = min(mBFloor, rssi);
66 }
67}

74void NoiseFloorDetector::processEvent(const Event &e)
75{
76 switch(e.type)
77 {
78 case CLOCK_EVENT:
79 if ( mTicks == 0xffffffff )
80 {
81 mTicks = 0;
82 }
83 else
84 {
85 ++mTicks;
86 }
87
88 if ( mTicks == 30 )
89 {
90 //DBG("Event pool utilization = %d, max = %d\r\n", EventPool::instance().utilization(), EventPool::instance().maxUtilization());
91 recalculate();
92 dump();
93 mTicks = 0;
94 }
95 break;
96 default:
97 break;
98 }
99}

101void NoiseFloorDetector::dump()
102{
103 Event *e = EventPool::instance().newEvent(PROPR_NMEA_SENTENCE);
104 if ( !e )
105 return;
106
107 sprintf(e->nmeaBuffer.sentence, "$PAINF,A,0x%.2x*", mChannelACurrent);
108 Utils::completeNMEA(e->nmeaBuffer.sentence);
109 EventQueue::instance().push(e);
110
111 e = EventPool::instance().newEvent(PROPR_NMEA_SENTENCE);
112 if ( !e )
113 return;
114
115 sprintf(e->nmeaBuffer.sentence, "$PAINF,B,0x%.2x*", mChannelBCurrent);
116 Utils::completeNMEA(e->nmeaBuffer.sentence);
117 EventQueue::instance().push(e);
118}

120void NoiseFloorDetector::recalculate()
121{
122 if ( mChannelACurrent == 0xff || mChannelBCurrent == 0xff )
123 {
124 mChannelACurrent = mAFloor;
125 mChannelBCurrent = mBFloor;
126 }
127 else
128 {
129 mChannelACurrent = (mChannelACurrent + mAFloor) / 2;
130 mChannelBCurrent = (mChannelBCurrent + mBFloor) / 2;
131 }
132
133 mAFloor = 0xff;
134 mBFloor = 0xff;
135}
