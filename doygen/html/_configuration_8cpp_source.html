70void Configuration::disableTX()

78bool Configuration::isTXEnabled()
79{
80 if ( mFlags.magic != CONFIG_FLAGS_MAGIC )
81 return true;
82
83 // Bit 0 in word 0 inhibits transmission
84 return (mFlags.flags[0] & 0x01) == 0;
85}

134void Configuration::reportSystemData()
135{
136 Event *e = EventPool::instance().newEvent(PROPR_NMEA_SENTENCE);
137 if ( !e )
138 return;
139
140 sprintf(e->nmeaBuffer.sentence, "$PAISYS,%s,%s,%s,%s,%s,%d*", hwRev(), FW_REV, serNum(), mcuType(),
141 breakoutType(), isBootloaderPresent());
142
143 Utils::completeNMEA(e->nmeaBuffer.sentence);
144 EventQueue::instance().push(e);
145}

147void Configuration::reportStationData()
148{
149 StationData d;
150 if ( !readStationData(d) )
151 memset(&d, 0, sizeof d);
152
153 Event *e = EventPool::instance().newEvent(PROPR_NMEA_SENTENCE);
154 if ( !e )
155 return;
156
157 sprintf(e->nmeaBuffer.sentence,
158 "$PAISTN,%lu,%s,%s,%d,%d,%d,%d,%d*",
159 d.mmsi,
160 d.name,
161 d.callsign,
162 d.type,
163 d.len,
164 d.beam,
165 d.portOffset,
166 d.bowOffset);
167
168 Utils::completeNMEA(e->nmeaBuffer.sentence);
169 EventQueue::instance().push(e);
170}

265#if OTP_DATA
266const OTPData *Configuration::readOTP()
267{
268 uint32_t address = nextAvailableOTPSlot();
269 if ( address == OTP_ADDRESS )
270 // There's nothing written!
271 return nullptr;
272
273 address -= sizeof(OTPData);
274 if ( IS_FLASH_OTP_ADDRESS(address) )
275 return (const OTPData*)address;
276
277 return nullptr;
278}
279
280bool Configuration::writeOTP(const OTPData &data)
281{
282 uint32_t address = nextAvailableOTPSlot();
283 if ( !IS_FLASH_OTP_ADDRESS(address) )
284 return false;
285
286 uint64_t *d = (uint64_t*)&data;
287
288 __HAL_FLASH_CLEAR_FLAG(FLASH_FLAG_ALL_ERRORS);
289 HAL_FLASH_Unlock();
290 HAL_StatusTypeDef status = HAL_OK;
291 for ( uint32_t dw = 0; dw < sizeof data/8; ++dw, ++d )
292 {
293 status = HAL_FLASH_Program(FLASH_TYPEPROGRAM_DOUBLEWORD, address + dw*8, *d);
294 if ( status != HAL_OK )
295 break;
296 }
297 HAL_FLASH_Lock();
298
299 return true;
300}
301
302uint32_t Configuration::nextAvailableOTPSlot()
303{
304 for ( uint32_t p = OTP_ADDRESS; p < OTP_ADDRESS+OTP_SIZE; p += sizeof (OTPData) )
305 {
306 OTPData *d = (OTPData*)p;
307 if ( d->magic == 0xFFFFFFFF )
308 return p;
309 }
310
311 return OTP_ADDRESS+OTP_SIZE;
312}
313#endif
