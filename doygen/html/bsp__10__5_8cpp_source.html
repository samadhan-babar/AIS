111

206void SystemClock_Config()
207{
208 RCC_OscInitTypeDef RCC_OscInitStruct;
209 RCC_ClkInitTypeDef RCC_ClkInitStruct;
210 RCC_PeriphCLKInitTypeDef PeriphClkInit;
211
214 RCC_OscInitStruct.OscillatorType = RCC_OSCILLATORTYPE_HSI;
215 RCC_OscInitStruct.HSIState = RCC_HSI_ON;
216 RCC_OscInitStruct.HSICalibrationValue = RCC_HSICALIBRATION_DEFAULT;
217 RCC_OscInitStruct.PLL.PLLState = RCC_PLL_ON;
218 RCC_OscInitStruct.PLL.PLLSource = RCC_PLLSOURCE_HSI;
219 RCC_OscInitStruct.PLL.PLLM = 1;
220 RCC_OscInitStruct.PLL.PLLN = 10; // 80 MHz
221#if defined(STM32L432xx) || defined(STM32L431xx)
222 RCC_OscInitStruct.PLL.PLLP = RCC_PLLP_DIV7;
223#endif
224 RCC_OscInitStruct.PLL.PLLQ = RCC_PLLQ_DIV2;
225 RCC_OscInitStruct.PLL.PLLR = RCC_PLLR_DIV2;
226 if (HAL_RCC_OscConfig(&RCC_OscInitStruct) != HAL_OK)
227 {
228 Error_Handler(0);
229 }
230
233 RCC_ClkInitStruct.ClockType = RCC_CLOCKTYPE_HCLK|RCC_CLOCKTYPE_SYSCLK
234 |RCC_CLOCKTYPE_PCLK1|RCC_CLOCKTYPE_PCLK2;
235 RCC_ClkInitStruct.SYSCLKSource = RCC_SYSCLKSOURCE_PLLCLK;
236 RCC_ClkInitStruct.AHBCLKDivider = RCC_SYSCLK_DIV1;
237 RCC_ClkInitStruct.APB1CLKDivider = RCC_HCLK_DIV1;
238 RCC_ClkInitStruct.APB2CLKDivider = RCC_HCLK_DIV1;
239
240 if (HAL_RCC_ClockConfig(&RCC_ClkInitStruct, FLASH_LATENCY_4) != HAL_OK)
241 {
242 Error_Handler(0);
243 }
244
245 PeriphClkInit.PeriphClockSelection = RCC_PERIPHCLK_USART1;
246 PeriphClkInit.Usart1ClockSelection = RCC_USART1CLKSOURCE_HSI;
247 if (HAL_RCCEx_PeriphCLKConfig(&PeriphClkInit) != HAL_OK)
248 {
249 Error_Handler(0);
250 }
251
254 if (HAL_PWREx_ControlVoltageScaling(PWR_REGULATOR_VOLTAGE_SCALE1) != HAL_OK)
255 {
256 Error_Handler(0);
257 }
258
261 HAL_SYSTICK_Config(HAL_RCC_GetHCLKFreq()/1000);
262
265 HAL_SYSTICK_CLKSourceConfig(SYSTICK_CLKSOURCE_HCLK);
266
267 /* SysTick_IRQn interrupt configuration */
268 HAL_NVIC_SetPriority(SysTick_IRQn, 0, 0);
269}

274 {

276 HAL_GPIO_Init(io->port, (GPIO_InitTypeDef*)&io->gpio);
277 if ( io->gpio.Mode == GPIO_MODE_OUTPUT_PP || io->gpio.Mode == GPIO_MODE_OUTPUT_OD )
278 {
279 HAL_GPIO_WritePin(io->port, io->gpio.Pin, io->init);
280 }

506void bsp_enter_dfu()
507{
508 // Cut off the GPS signals immediately to prevent its UART from transmitting and hijacking the bootloader upon reset
509 bsp_gnss_off();
510
511 HAL_Delay(1000);
512
513 // This flag simply tells main() to jump to the ROM bootloader immediately upon reset, before initializing anything
514 *(uint32_t*)BOOTMODE_ADDRESS = DFU_FLAG_MAGIC;
515
516 bsp_reboot();
517}
