80bool RFIC::sendCmd(uint8_t cmd, void* params, uint8_t paramLen, void* result, uint8_t resultLen)
81{
82 if ( mCTSPending )
83 {
84 while ( readSPIResponse(NULL, 0) == false)
85 ;
86 mCTSPending = false;
87 }
88
89 //bsp_signal_high();
90 spiOn();
91
92 bsp_tx_spi_byte(cmd);
93
94 if ( params )
95 {
96 uint8_t *b = (uint8_t*) params;
97 for ( int i = 0; i < paramLen; ++i )
98 {
99 bsp_tx_spi_byte(b[i]);
100 }
101 }
102 spiOff();
103 //bsp_signal_low();
104
105 //bsp_signal_high();
106 while ( readSPIResponse(result, resultLen) == false)
107 ;
108
109 //bsp_signal_low();
110 return true;
111}

139bool RFIC::readSPIResponse(void *data, uint8_t length)
140{
141 spiOn();
142 bsp_tx_spi_byte(0x44);
143 if ( bsp_tx_spi_byte(0) != 0xff )
144 {
145 spiOff();
146 return false;
147 }
148
149 if ( data )
150 {
151 uint8_t* b = (uint8_t*) data;
152 uint8_t i = 0;
153 while (i < length)
154 {
155 b[i] = bsp_tx_spi_byte(0);
156 ++i;
157 }
158 }
159
160 spiOff();
161 return true;
162}

164void RFIC::configure()
165{
166 PART_INFO_REPLY reply;
167 sendCmd(PART_INFO, nullptr, 0, &reply, sizeof reply);
168 mPartNumber = reply.PartNumberH << 8 | reply.PartNumberL;
169
170 uint8_t *radio_configuration = nullptr;
171 switch(mPartNumber)
172 {
173 case 0x4467:
174 radio_configuration = get_si4467_config_array();
175 break;
176 case 0x4460:
177 radio_configuration = get_si4460_config_array();
178 break;
179 default:
180 radio_configuration = get_si4463_config_array();
181 }
182
183 uint8_t *cfg = radio_configuration;
184 while (*cfg)
185 { // configuration array stops with 0
186 uint8_t count = (*cfg++) - 1; // 1st byte: number of bytes, incl. command
187 uint8_t cmd = *cfg++; // 2nd byte: command
188 sendCmd(cmd, cfg, count, NULL, 0); // send bytes to chip
189 cfg += count; // point at next line
190 }
191
192 if ( Configuration::instance().isXOTrimmed() )
193 {
194 setXOTrimValue(Configuration::instance().getXOTrimValue());
195 }
196}

215bool RFIC::isInitialized()
216{
217 HAL_GPIO_WritePin(mSDNP, mSDNPin, GPIO_PIN_SET);
218 HAL_Delay(10);
219 HAL_GPIO_WritePin(mSDNP, mSDNPin, GPIO_PIN_RESET);
220 HAL_Delay(100);
221
222 CHIP_STATUS_REPLY chip_status;
223 memset(&chip_status, 0, sizeof chip_status);
224 sendCmd(GET_CHIP_STATUS, NULL, 0, &chip_status, sizeof chip_status);
225 if ( chip_status.Current & 0x08 )
226 {
227 return false;
228 }
229 else
230 {
231 return true;
232 }
233}

235bool RFIC::powerOnReset()
236{
237 // Pull SDN high to shut down the IC
238 HAL_GPIO_WritePin(mSDNP, mSDNPin, GPIO_PIN_SET);
239
240 // Wait at least 10 microseconds
241 HAL_Delay(1);
242
243 // Pull SDN low and poll the status of GPIO1
244 HAL_GPIO_WritePin(mSDNP, mSDNPin, GPIO_PIN_RESET);
245
246 uint32_t start = HAL_GetTick();
247 while ( HAL_GetTick() - start < 1000 )
248 {
249 if ( HAL_GPIO_ReadPin(mDataPort, mDataPin) == GPIO_PIN_SET )
250 return true;
251 }
252
253 return false;
254}
