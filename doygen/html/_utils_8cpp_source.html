86void Utils::trim(string &str)
87{
88 static string whitespace(" \t\f\v\n\r");
89 string::size_type pos = str.find_last_not_of(whitespace);
90 if ( pos != string::npos )
91 {
92 str.erase(pos + 1);
93 pos = str.find_first_not_of(whitespace);
94 if ( pos != string::npos )
95 str.erase(0, pos);
96 }
97 else
98 str.erase(str.begin(), str.end());
99}

109float Utils::latitudeFromNMEA(const string &decimal, const string &hemisphere)
110{
111 // Latitude always starts with 4 integers: 2 for degrees, 2 for minutes . N decimal minutes
112 string degStr = decimal.substr(0, 2);
113 string decStr = decimal.substr(2);
114
115 float deg = atof(degStr.c_str());
116 float min = atof(decStr.c_str());
117 return (hemisphere == "N" ? 1 : -1) * (deg + min / 60.0f);
118}

131uint16_t Utils::reverseBits(uint16_t crc)
132{
133 uint16_t result = 0;
134 for ( size_t i = 0; i < 16; ++i )
135 {
136 result <<= 1;
137 result |= ((crc & (1 << i)) >> i);
138 }
139
140 return result;
141}

143uint16_t Utils::crc16(uint8_t *bytes, uint16_t length)
144{
145 uint16_t crc = 0xffff;
146 for ( uint16_t b = 0; b < length; ++b )
147 {
148 crc = ((crc << 8) & 0xff00) ^ CRC16_XMODEM_TABLE[((crc >> 8) & 0xff) ^ bytes[b]];
149 }
150 return Utils::reverseBits(~crc);
151}

153void Utils::tokenize(const string &str, char delim, vector<string> &result)
154{
155 if ( str.empty() )
156 return;
157
158 string token;
159 result.clear();
160 for ( unsigned i = 0; i < str.length(); ++i )
161 {
162 if ( str[i] == delim )
163 {
164 result.push_back(token);
165 token.erase();
166 }
167 else
168 token += str[i];
169 }
170
171 if ( !token.empty() )
172 result.push_back(token);
173}

176uint32_t Utils::coordinateToUINT32(double value)
177{

185float Utils::coordinateFromUINT32(uint32_t value, uint8_t numBits)
186{
187 if ( value & (1 << (numBits - 1)) )
188 {
189 value = ~value;
190 for ( int i = numBits - 1; i < 32; ++i )
191 value &= ~(1 << i);
192
193 return value / -600000.0f;
194 }
195 else
196 {
197 return value / 600000.0f;
198 }
199}
