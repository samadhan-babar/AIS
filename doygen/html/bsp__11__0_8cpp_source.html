219void SystemClock_Config()
220{
221 RCC_OscInitTypeDef RCC_OscInitStruct;
222 RCC_ClkInitTypeDef RCC_ClkInitStruct;
223 RCC_PeriphCLKInitTypeDef PeriphClkInit;
224
227 RCC_OscInitStruct.OscillatorType = RCC_OSCILLATORTYPE_HSI;
228 RCC_OscInitStruct.HSIState = RCC_HSI_ON;
229 RCC_OscInitStruct.HSICalibrationValue = RCC_HSICALIBRATION_DEFAULT;
230 RCC_OscInitStruct.PLL.PLLState = RCC_PLL_ON;
231 RCC_OscInitStruct.PLL.PLLSource = RCC_PLLSOURCE_HSI;
232 RCC_OscInitStruct.PLL.PLLM = 1;
233 RCC_OscInitStruct.PLL.PLLN = 10; // 80 MHz
234#if defined(STM32L432xx) || defined(STM32L431xx)
235 RCC_OscInitStruct.PLL.PLLP = RCC_PLLP_DIV7;
236#endif
237 RCC_OscInitStruct.PLL.PLLQ = RCC_PLLQ_DIV2;
238 RCC_OscInitStruct.PLL.PLLR = RCC_PLLR_DIV2;
239 if (HAL_RCC_OscConfig(&RCC_OscInitStruct) != HAL_OK)
240 {
241 Error_Handler(0);
242 }
243
246 RCC_ClkInitStruct.ClockType = RCC_CLOCKTYPE_HCLK|RCC_CLOCKTYPE_SYSCLK
247 |RCC_CLOCKTYPE_PCLK1|RCC_CLOCKTYPE_PCLK2;
248 RCC_ClkInitStruct.SYSCLKSource = RCC_SYSCLKSOURCE_PLLCLK;
249 RCC_ClkInitStruct.AHBCLKDivider = RCC_SYSCLK_DIV1;
250 RCC_ClkInitStruct.APB1CLKDivider = RCC_HCLK_DIV1;
251 RCC_ClkInitStruct.APB2CLKDivider = RCC_HCLK_DIV1;
252
253 if (HAL_RCC_ClockConfig(&RCC_ClkInitStruct, FLASH_LATENCY_4) != HAL_OK)
254 {
255 Error_Handler(0);
256 }
257
258 PeriphClkInit.PeriphClockSelection = RCC_PERIPHCLK_USART1;
259 PeriphClkInit.Usart1ClockSelection = RCC_USART1CLKSOURCE_HSI;
260 if (HAL_RCCEx_PeriphCLKConfig(&PeriphClkInit) != HAL_OK)
261 {
262 Error_Handler(0);
263 }
264
267 if (HAL_PWREx_ControlVoltageScaling(PWR_REGULATOR_VOLTAGE_SCALE1) != HAL_OK)
268 {
269 Error_Handler(0);
270 }
271
274 HAL_SYSTICK_Config(HAL_RCC_GetHCLKFreq()/1000);
275
278 HAL_SYSTICK_CLKSourceConfig(SYSTICK_CLKSOURCE_HCLK);
279
280 /* SysTick_IRQn interrupt configuration */
281 HAL_NVIC_SetPriority(SysTick_IRQn, 0, 0);
282}

284void gpio_pin_init()
285{
286 for ( unsigned i = 0; i < sizeof __gpios / sizeof(GPIO); ++i )
287 {
288 const GPIO* io = &__gpios[i];
289 if ( io->gpio.Mode == GPIO_MODE_OUTPUT_PP || io->gpio.Mode == GPIO_MODE_OUTPUT_OD )
290 {
291 HAL_GPIO_WritePin(io->port, io->gpio.Pin, io->init);
292 }
293 HAL_GPIO_Init(io->port, (GPIO_InitTypeDef*)&io->gpio);
294 }

344void bsp_write_station_data(const StationData &data)
345{
346 bsp_erase_station_data();
347 StationDataPage page;
348 page.station = data;
349 page.station.magic = STATION_DATA_MAGIC;
350
351 uint32_t pageAddress = STATION_DATA_ADDRESS;
352 __HAL_FLASH_CLEAR_FLAG(FLASH_FLAG_ALL_ERRORS);
353 HAL_FLASH_Unlock();
354 HAL_StatusTypeDef status = HAL_OK;
355 for ( uint32_t dw = 0; dw < sizeof page/8; ++dw )
356 {
357 status = HAL_FLASH_Program(FLASH_TYPEPROGRAM_DOUBLEWORD, pageAddress + dw*8, page.dw[dw]);
358 if ( status != HAL_OK )
359 break;
360 }
361 HAL_FLASH_Lock();
362}
363
364void bsp_erase_station_data()
365{
366 uint32_t page = (STATION_DATA_ADDRESS - FLASH_BASE) / FLASH_PAGE_SIZE;
367
368 __HAL_FLASH_CLEAR_FLAG(FLASH_FLAG_ALL_ERRORS);
369 if ( HAL_FLASH_Unlock() != HAL_OK )
370 return;
371
372 FLASH_EraseInitTypeDef erase;
373 erase.TypeErase = FLASH_TYPEERASE_PAGES;
374 erase.Banks = FLASH_BANK_1;
375 erase.Page = page;
376 erase.NbPages = 1;
377
378 uint32_t errPage;
379 HAL_FLASHEx_Erase(&erase, &errPage);
380 HAL_FLASH_Lock();
381}

601void bsp_enter_dfu()
602{
603 // Cut off the GPS signals immediately to prevent its UART from transmitting and hijacking the bootloader upon reset
604 bsp_gnss_off();
605
606 HAL_Delay(1000);
607
608 // This flag simply tells main() to jump to the ROM bootloader immediately upon reset, before initializing anything
609 *(uint32_t*)BOOTMODE_ADDRESS = DFU_FLAG_MAGIC;
610
611 bsp_reboot();
612}
