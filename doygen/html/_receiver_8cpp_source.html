112void Receiver::onBitClock()
113{
114 ++mSlotBitNumber;
115
116 // Don't waste time processing bits when the transceiver is transmitting
117 if ( gRadioState == RADIO_TRANSMITTING )
118 {
119 return;
120 }
121
122
123 if ( !mRXPacket )
124 {
125 mRXPacket = EventPool::instance().newRXPacket();
126 if ( !mRXPacket )
127 {
128 return;
129 }
130 }
131
132
133 uint8_t bit = HAL_GPIO_ReadPin(mDataPort, mDataPin);
134 Receiver::Action action = processNRZIBit(bit);
135 if ( action == RESTART_RX )
136 {
137 startReceiving(mNextChannel, false);
138 }
139 else if ( mTimeSlot != 0xffffffff && mSlotBitNumber != 0xffff && mSlotBitNumber == CCA_SLOT_BIT )
140 {
141 mRXPacket->setRSSI(reportRSSI());
142 }
143}

150void Receiver::timeSlotStarted(uint32_t slot)
151{
152 // This should never be called while transmitting. Transmissions start after the slot boundary and end before the end of it.
153 ASSERT(gRadioState == RADIO_RECEIVING);
154
155 mSlotBitNumber = -1;
156 mTimeSlot = slot;
157 if ( mBitState == BIT_STATE_IN_PACKET )
158 return;
159
160 if ( mRXPacket )
161 mRXPacket->setSlot(slot);
162
163 if ( mChannel != mNextChannel )
164 {
165 startReceiving(mNextChannel, false);
166 }
167}

172Receiver::Action Receiver::processNRZIBit(uint8_t bit)
173{
174 if ( mLastNRZIBit == 0xff )
175 {
176 mLastNRZIBit = bit;
177 return NO_ACTION;
178 }
179
180 uint8_t decodedBit = !(mLastNRZIBit ^ bit);
181
182 switch (mBitState) {
183 case BIT_STATE_PREAMBLE_SYNC:
184 {
185 mLastNRZIBit = bit;
186 mBitWindow <<= 1;
187 mBitWindow |= decodedBit;
188
189 /*
190 * By checking for the last few preamble bits plus the HDLC start flag,
191 * we gain enough confidence that this is not random noise.
192 */
193 //if ( mBitWindow == 0b1010101001111110 || mBitWindow == 0b0101010101111110 )
194 if ( (mBitWindow & 0x0fff) == 0b101001111110 || (mBitWindow &0x0fff) == 0b010101111110 )
195 {
196 mBitState = BIT_STATE_IN_PACKET;
197 mRXPacket->setChannel(mChannel);
198 }
199
200 break;
201 }
202 case BIT_STATE_IN_PACKET:
203 {
204 if ( mOneBitCount >= 7 || mRXPacket->size() >= MAX_AIS_RX_PACKET_SIZE - 2 )
205 {
206 // Start over
207 return RESTART_RX;
208 }
209
210 mLastNRZIBit = bit;
211 mBitWindow <<= 1;
212 mBitWindow |= decodedBit;
213
214 if ( (mBitWindow & 0x00ff) == 0x7E )
215 {
216 // We have a complete packet
217 mBitState = BIT_STATE_PREAMBLE_SYNC;
222 pushPacket();
223 return RESTART_RX;
224 }
225 else
226 {
227 addBit(decodedBit);
228 }
229
230 break;
231 }
232 }
233
234 return NO_ACTION;
235}

238bool Receiver::addBit(uint8_t bit)
239{
240 bool result = true;
241 if ( bit )
242 {
243 ++mOneBitCount;
244 }
245 else
246 {
247 // Don't put stuffed bits into the packet
248 if ( mOneBitCount == 5 )
249 result = false;
250
251 mOneBitCount = 0;
252 }
253
254 if ( result )
255 {
256 mRXByte <<= 1;
257 mRXByte |= bit;
258 ++mBitCount;
259 }
260
261 if ( mBitCount == 8 )
262 {
263 // Commit to the packet!
264 mRXPacket->addByte(mRXByte);
265 mBitCount = 0;
266 mRXByte = 0;
267 }
268
269 return result;
270}

318{

321 gpiocfg.GPIO1 = 0x14; // RX data bits
322 gpiocfg.GPIO2 = 0x00; // No change
323 gpiocfg.GPIO3 = 0x11; // RX data clock

326 gpiocfg.GENCFG = 0x00; // No change
327 sendCmd(GPIO_PIN_CFG, &gpiocfg, sizeof gpiocfg, NULL, 0);
