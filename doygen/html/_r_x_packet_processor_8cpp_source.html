52void RXPacketProcessor::processEvent(const Event &e)
53{
54 switch(e.type)
55 {
56 case AIS_PACKET_EVENT:
57 {
58 ASSERT(e.rxPacket);
59 if (e.rxPacket->isBad() || !e.rxPacket->checkCRC ())
60 return;
61
62 bsp_rx_led_on();
63
64 if ( e.rxPacket->messageType() == 15 )
65 {
66 AISMessage15 msg;
67 if ( msg.decode(*e.rxPacket) )
68 {
69 // Make sure we actually can transmit something
70 if ( mStationData.magic != STATION_DATA_MAGIC )
71 break;
72
73 // This is an interrogation. If we are a target, push an appropriate event into the queue
74
75 // It is possible that we are the target for more than one type of message (18 + 24)
76 for ( uint8_t i = 0; i < 3; ++i )
77 {
78 AISMessage15::InterrogationTarget &target = msg.targets[i];
79 if ( target.mmsi == mStationData.mmsi )
80 {
81 switch(target.messageType)
82 {
83 case 18:
84 case 24:
85 {
86#if 0
87 Event ie(INTERROGATION_EVENT);
88 ie.interrogation.channel = e.rxPacket.channel();
89 ie.interrogation.messageType = target.messageType;
90
91 //printf2("Scheduling message %d in response to interrogation\r\n", ie->interrogation.messageType);
92 EventQueue::instance().push(ie);
93#endif
94 break;
95 }
96 default:
97 // Why do base stations sometimes request message 5 from class B transponders?
98 //DBG("Ignoring malformed message 15 from MMSI %d\r\n", e.rxPacket.mmsi());
99 break;
100 }
101 }
102 }
103
104 }
105 } // If message 15
106
107
108 mSentences.clear();
109
110 ASSERT_VALID_PTR(e.rxPacket);
111 mEncoder.encode(*(e.rxPacket), mSentences);
112 for (vector<string>::iterator i = mSentences.begin(); i != mSentences.end(); ++i)
113 {
114#ifdef MULTIPLEXED_OUTPUT
115 sprintf(__buff, "%s\r\n", i->c_str());
116 DataTerminal::instance().write("NMEA", __buff);
117#else
118 DataTerminal::instance().write(i->c_str());
119 DataTerminal::instance().write("\r\n");
120#endif
121 }
122
123
124 // Special handling for specific messages that we care about
125 switch (e.rxPacket->messageType())
126 {
127 case 20:
128 // TODO: This is a time slot reservation from a base station. Use this information to block those time slots.
129 break;
130 case 22:
131 /*
132 TODO:
133
134 This is the frequency management message. Since we support all of the upper VHF band (161.500 MHz - 162.025 MHz),
135 we use this to switch our primary 2 channels where instructed. That said, I'm not completely clear on how channels

137 but whoever wrote this document did not care about comprehension :(

142 break;
143 case 23:
144 /*
145 TODO: This is the group assignment message. Base stations can use this to configure our transmission interval
146 or enforce silent periods as well. We should comply.
147 */
148 break;
149 }

153 break;
154 }
155 default:
156 break;
157 }
