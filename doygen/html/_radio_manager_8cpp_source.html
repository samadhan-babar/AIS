51void RadioManager::init()
52{
53 NoiseFloorDetector::instance();
54 mTransceiverIC = new Transceiver(SDN1_PORT, SDN1_PIN,
55 CS1_PORT, CS1_PIN,
56 TRX_IC_DATA_PORT, TRX_IC_DATA_PIN,
57 TRX_IC_CLK_PORT, TRX_IC_CLK_PIN, 0);
58
59 if ( mTransceiverIC->isResponsive() )
60 {
61 mTransceiverIC->init();
62 }
63 else
64 {
65 reportError(1);
66 }
67
68 mReceiverIC = new Receiver(SDN2_PORT, SDN2_PIN,
69 CS2_PORT, CS2_PIN,
70 RX_IC_DATA_PORT, RX_IC_DATA_PIN,
71 RX_IC_CLK_PORT, RX_IC_CLK_PIN, 1);
72 if ( mReceiverIC->isResponsive() )
73 {
74 mReceiverIC->init();
75 }
76 else
77 {
78 reportError(2);
79 }
80
81 if ( mReceiverIC->isResponsive() && mTransceiverIC->isResponsive() )
82 mInitializing = false;
83}

140void RadioManager::processEvent(const Event &e)
141{
142 mUTC = e.clock.utc;
143
144 if ( mStartTime == 0 )
145 {
146 mStartTime = mUTC;
147 }
148
149 if ( !mTXQueue.empty() && mTransceiverIC->assignedTXPacket() == NULL )
150 {
151 // There is no current TX operation pending, so we assign one
152 TXPacket *packet = NULL;
153 mTXQueue.pop(packet);
154 ASSERT(packet);
155
156 VHFChannel txChannel = packet->channel();
157
158 // Do we need to swap channels?
159 if ( txChannel != mTransceiverIC->channel() )
160 {
161 // The receiver needs to be explicitly told to switch channels
162 if ( mReceiverIC )
163 mReceiverIC->switchToChannel(alternateChannel(txChannel));
164 }
165
166 // The transceiver will switch channel if the packet channel is different
167 mTransceiverIC->assignTXPacket(packet);
168 }
169 else if ( mUTC - mStartTime >= 3600 )
170 {
177 bsp_reboot();
178 }
179
180}

208void RadioManager::scheduleTransmission(TXPacket *packet)
209{
210#if REPORT_TX_SCHEDULING
211 Event *e = EventPool::instance().newEvent(PROPR_NMEA_SENTENCE);
212#endif
213 if ( !mTXQueue.push(packet) )
214 {
215#if REPORT_TX_SCHEDULING
216 if ( e )
217 {
218 sprintf(e->nmeaBuffer.sentence, "$PAISCHTX,%s,%d*", packet->messageType(), TX_QUEUE_FULL);
219 Utils::completeNMEA(e->nmeaBuffer.sentence);
220 EventQueue::instance().push(e);
221 }
222#endif
223 TXPacketPool::instance().deleteTXPacket(packet);
224 }
225#if REPORT_TX_SCHEDULING
226 else
227 {
228 if ( e )
229 {
230 sprintf(e->nmeaBuffer.sentence, "$PAISCHTX,%s,%d*", packet->messageType(), TX_NO_ERROR);
231 Utils::completeNMEA(e->nmeaBuffer.sentence);
232 EventQueue::instance().push(e);
233 }
234 }
235#endif
236}
