189void SystemInit(void)
190{
191 /* FPU settings ------------------------------------------------------------*/
192 #if (__FPU_PRESENT == 1) && (__FPU_USED == 1)
193 SCB->CPACR |= ((3UL << 10*2)|(3UL << 11*2)); /* set CP10 and CP11 Full Access */
194 #endif
195
196 /* Reset the RCC clock configuration to the default reset state ------------*/
197 /* Set MSION bit */
198 RCC->CR |= RCC_CR_MSION;
199
200 /* Reset CFGR register */
201 RCC->CFGR = 0x00000000U;
202
203 /* Reset HSEON, CSSON , HSION, and PLLON bits */
204 RCC->CR &= 0xEAF6FFFFU;
205
206 /* Reset PLLCFGR register */
207 RCC->PLLCFGR = 0x00001000U;
208
209 /* Reset HSEBYP bit */
210 RCC->CR &= 0xFFFBFFFFU;
211
212 /* Disable all interrupts */
213 RCC->CIER = 0x00000000U;
214
215 /* Configure the Vector Table location add offset address ------------------*/
216#ifdef VECT_TAB_SRAM
217 SCB->VTOR = SRAM_BASE | VECT_TAB_OFFSET; /* Vector Table Relocation in Internal SRAM */
218#else
219 SCB->VTOR = FLASH_BASE | VECT_TAB_OFFSET; /* Vector Table Relocation in Internal FLASH */
220#endif
221}

265void SystemCoreClockUpdate(void)
266{
267 uint32_t tmp = 0U, msirange = 0U, pllvco = 0U, pllr = 2U, pllsource = 0U, pllm = 2U;
268
269 /* Get MSI Range frequency--------------------------------------------------*/
270 if((RCC->CR & RCC_CR_MSIRGSEL) == RESET)
271 { /* MSISRANGE from RCC_CSR applies */
272 msirange = (RCC->CSR & RCC_CSR_MSISRANGE) >> 8U;
273 }
274 else
275 { /* MSIRANGE from RCC_CR applies */
276 msirange = (RCC->CR & RCC_CR_MSIRANGE) >> 4U;
277 }
278 /*MSI frequency range in HZ*/
279 msirange = MSIRangeTable[msirange];
280
281 /* Get SYSCLK source -------------------------------------------------------*/
282 switch (RCC->CFGR & RCC_CFGR_SWS)
283 {
284 case 0x00: /* MSI used as system clock source */
285 SystemCoreClock = msirange;
286 break;
287
288 case 0x04: /* HSI used as system clock source */
289 SystemCoreClock = HSI_VALUE;
290 break;
291
292 case 0x08: /* HSE used as system clock source */
293 SystemCoreClock = HSE_VALUE;
294 break;
295
296 case 0x0C: /* PLL used as system clock source */
297 /* PLL_VCO = (HSE_VALUE or HSI_VALUE or MSI_VALUE/ PLLM) * PLLN
298 SYSCLK = PLL_VCO / PLLR
299 */
300 pllsource = (RCC->PLLCFGR & RCC_PLLCFGR_PLLSRC);
301 pllm = ((RCC->PLLCFGR & RCC_PLLCFGR_PLLM) >> 4U) + 1U ;
302
303 switch (pllsource)
304 {
305 case 0x02: /* HSI used as PLL clock source */
306 pllvco = (HSI_VALUE / pllm);
307 break;
308
309 case 0x03: /* HSE used as PLL clock source */
310 pllvco = (HSE_VALUE / pllm);
311 break;
312
313 default: /* MSI used as PLL clock source */
314 pllvco = (msirange / pllm);
315 break;
316 }
317 pllvco = pllvco * ((RCC->PLLCFGR & RCC_PLLCFGR_PLLN) >> 8U);
318 pllr = (((RCC->PLLCFGR & RCC_PLLCFGR_PLLR) >> 25U) + 1U) * 2U;
319 SystemCoreClock = pllvco/pllr;
320 break;
321
322 default:
323 SystemCoreClock = msirange;
324 break;
325 }
326 /* Compute HCLK clock frequency --------------------------------------------*/
327 /* Get HCLK prescaler */
328 tmp = AHBPrescTable[((RCC->CFGR & RCC_CFGR_HPRE) >> 4U)];
329 /* HCLK clock frequency */
330 SystemCoreClock >>= tmp;
331}
