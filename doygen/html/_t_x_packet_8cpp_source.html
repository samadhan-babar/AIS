96void TXPacket::addBit(uint8_t bit)
97{
98 ASSERT(mSize < MAX_AIS_TX_PACKET_SIZE);
99
100 uint16_t index = mSize / 8;
101 uint8_t offset = mSize % 8;
102
103 if ( bit )
104 mPacket[index] |= ( 1 << offset );
105
106 ++mSize;
107}
108
109void TXPacket::pad()
110{
111 uint16_t rem = 8 - mSize % 8;
112 for ( uint16_t i = 0; i < rem; ++i )
113 addBit(0);
114}

129uint8_t TXPacket::nextBit()
130{
131 ASSERT(mPosition < mSize);
132 if ( mTestPacket )
133 {
134 ++mPosition;
135 return rand() % 2;
136 }
137 else
138 {
139 uint16_t index = mPosition / 8;
140 uint8_t offset = mPosition % 8;
141 ++mPosition;
142 return (mPacket[index] & ( 1 << offset )) != 0;
143 }
144}
