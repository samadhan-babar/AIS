34{

37 ObjectPool<T>(uint32_t size)
38 : mQueue(size)
39 {
40 mSize = size;
41 mUtilization = 0;
42 mMaxUtilization = 0;
43
44 //printf_serial_now("ObjectPool @%p\r\n", this);
45 for ( uint32_t i = 0; i < mSize; ++i )
46 {
47 T *p = new T();
48 ASSERT_VALID_PTR(p);
49 mQueue.push(p);
50 //printf_serial_now("\t@%p\r\n", p);
51 }
52 }
53
54 T *get()

59 }
