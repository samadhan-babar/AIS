49void DataTerminal::processEvent(const Event &e)

99void termInputCB(char c)
100{
101 if ( c == '\r' )
102 {
103 return;
104 }
105
106 __rxbuff[__rxpos++] = c;
107 if ( __rxpos >= sizeof __rxbuff )
108 {
109 __rxpos = 0;
110 }
111 else if ( c == '\n' )
112 {
113 __rxbuff[__rxpos++] = 0;
114 Event *e = EventPool::instance().newEvent(COMMAND_EVENT);
115 strlcpy(e->command.buffer, __rxbuff, sizeof e->command.buffer);
116 EventQueue::instance().push(e);
117 __rxpos = 0;
118 }
119}
