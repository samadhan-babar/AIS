85void CommandProcessor::processCommand(const char *buff)
86{
87 string s(buff);
88 Utils::trim(s);
89
90 if ( s.find("station ") == 0 )
91 {
92 /*
93 * The station command format is:
94 * station mmsi,name,callsign,type,len,beam,portoffset,bowoffset
95 */
96
97 StationData station;
98
99 string params = s.substr(8);
100 if (params.empty())
101 return;
102
103 vector<string> tokens;
104 Utils::tokenize(params, ',', tokens);
105 if ( tokens.size() < 8 )
106 return;
107
108 memset(&station, 0, sizeof station);
109 station.mmsi = Utils::toInt(tokens[0]);
110 strlcpy(station.name, tokens[1].c_str(), sizeof station.name);
111 strlcpy(station.callsign, tokens[2].c_str(), sizeof station.callsign);
112 int type = (VesselType)Utils::toInt(tokens[3]);
113 if ( type == 30 || type == 34 || type == 36 || type == 37 )
114 station.type = (VesselType)type;
115 station.len = Utils::toInt(tokens[4]);
116 station.beam = Utils::toInt(tokens[5]);
117 station.portOffset = Utils::toInt(tokens[6]);
118 station.bowOffset = Utils::toInt(tokens[7]);
119 station.magic = STATION_DATA_MAGIC;
120
121 Configuration::instance().writeStationData(station);
122 }
123 else if ( s.find("station?") == 0 )
124 {
125 Configuration::instance().reportStationData();
126 }
127 else if ( s.find("sys?") == 0 )
128 {
129 Configuration::instance().reportSystemData();
130 }
131 else if ( s.find("dfu") == 0 )
132 {
133 jumpToBootloader();
134 }
135 else if ( s.find("erase station") == 0 )
136 {
137 Configuration::instance().eraseStationData();
138 Configuration::instance().reportStationData();
139 }
140 else if ( s.find("factory reset") == 0 )
141 {
142 Configuration::instance().factoryReset();
143 Configuration::instance().reportStationData();
144 }
145 else if ( s.find("tx test") == 0 )
146 {
147 fireTestPacket();
148 }
149 else if ( s.find("tx on") == 0 )
150 {
151 Configuration::instance().enableTX();
152 TXScheduler::instance().reportTXStatus();
153 }
154 else if ( s.find("tx off") == 0 )
155 {
156 Configuration::instance().disableTX();
157 TXScheduler::instance().reportTXStatus();
158 }
159 else if ( s.find("tx?") == 0 )
160 {
161 TXScheduler::instance().reportTXStatus();
162 }
163 else if ( s.find("txcw a") == 0 )
164 {
165 RadioManager::instance().transmitCW(CH_87);
166 }
167 else if ( s.find("txcw b") == 0 )
168 {
169 RadioManager::instance().transmitCW(CH_88);
170 }
171 else if ( s.find("stoptx") == 0 )
172 {
173 RadioManager::instance().stopTX();
174 }
175 else if (s.find("reboot") == 0 )
176 {
177 bsp_reboot();
178 }
179 else if ( s.find("cli") == 0 )
180 {
181 enterCLIMode();
182 }
183 else if ( s.find("gps off") == 0 )
184 {
185 bsp_gnss_off();
186 }
187 else if ( s.find("gps on") == 0 )
188 {
189 bsp_gnss_on();
190 }
191 else if ( s.find("msg24") == 0 )
192 {
193 TXScheduler::instance().queueMessage24(CH_87);
194 }
195#if OTP_DATA
196 else if ( s.find("otp?") == 0 )
197 {
198 dumpOTPData();
199 }
200 else if ( s.find("otp ") == 0 )
201 {
202 writeOTPData(s);
203 }
204#endif
205 else if ( s.find("xotrim ") == 0 )
206 {
207 uint8_t value = Utils::toInt(s.substr(7));
208 RadioManager::instance().setXOTrimValue(value);
209 Configuration::instance().setXOTrimValue(value);
210 }
211 else if ( s.find("xotrim?") == 0 )
212 {
213 Configuration::instance().reportXOTrimValue();
214 }
215}

234void CommandProcessor::writeOTPData(const std::string &s)
235{
236 string params = s.substr(4);
237 if (params.empty())
238 return;
239
240 vector<string> tokens;
241 Utils::tokenize(params, ' ', tokens);
242
243 OTPData data;
244 memset(&data, 0, sizeof data);
245
246 data.magic = OTP_MAGIC;
247 data.rev = OTP_REV;
248 strlcpy(data.hwrev, tokens[0].c_str(), sizeof data.hwrev);
249
250 if ( tokens.size() > 2 )
251 strlcpy(data.serialnum, tokens[1].c_str(), sizeof data.serialnum);
252
253 bool result = Configuration::instance().writeOTP(data);
254 if ( result )
255 dumpOTPData();
256}
257#endif
